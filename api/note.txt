==============================================================
RUN PROJECT

// Lancer le projet avec Nodemon
npm run dev (auto refresh)


==============================================================
PRISMA

// Importer les donnes de la DB vers les schema du projet
npx prisma db pull

// Nettoie tout et reset la base
npx prisma migrate reset

// Migrer les schema vers la DB
npx prisma migrate dev

// Feed les tables avec les seeds
npx prisma db seed

==============================================================
TEST

// Lancer un test
node --test tests/log.test.js

// Lancer tous les tests
node --test

// Pour exécuter uniquement certains tests par nom 
node --test --test-name "addition works"
